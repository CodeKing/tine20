from __future__ import with_statement
import unittest

from sikuli.Sikuli import *

class Chromium(object):

    def __init__(self, url):
        self.url = url

    def __enter__(self):
        Screen(0)
        app = App.open('chromium-browser')
        wait(2)
        type("%s\n" % self.url)
        wait(2)

    def __exit__(self, type_, value, traceback):
        type(Key.F4, KEY_ALT)
